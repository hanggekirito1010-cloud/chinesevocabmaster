// Backend Endpoint Example (Node.js/Express)
app.post('/api/log-payment', async (req, res) => {
    try {
        const { email, amount, paymentMethod, timestamp } = req.body;
        
        // 1. Save payment record to your database
        await db.collection('payments').insertOne({
            email,
            amount,
            paymentMethod,
            timestamp,
            status: 'completed'
        });
        
        // 2. Call Quizlet API to invite user
        await quizletAPI.inviteToSet({
            setId: 'YOUR_QUIZLET_SET_ID',
            email: email
        });
        
        res.status(200).json({ success: true, message: 'Payment logged and invitation sent' });
    } catch (error) {
        console.error('Payment processing error:', error);
        res.status(500).json({ success: false, message: 'Failed to process payment' });
    }
});
